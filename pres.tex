\section{Algorithms}
\subsection{Sample algorithm}
\begin{frame}
	\frametitle{Algorithm}
	\begin{algorithm}[H]
\fontsize{10pt}{12}\selectfont
\caption{Sum of first $N$ natural numbers}
\begin{algorithmic}[1]
\State $S \leftarrow 0$
\For {$i=1$ to $N$}
\State $ S \leftarrow S + i $
\EndFor{}
\State Output $S$
\end{algorithmic}
\end{algorithm}
\end{frame}
